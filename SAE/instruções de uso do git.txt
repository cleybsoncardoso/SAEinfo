## configura o nome no git
git config --global user.name "seu nome"



##configura o email
git config --global user.email "seu email"



##realizar o seguinte comando dentro da pasta que você escolher como repositório
git init



##adicionando o repositório de origem
git remote add origin https://github.com/cleybsoncardoso/SAEinfo.git



##baixando o repositório de origem
git pull origin master



##para fazer o upload das mudanças deve-se primeiro:
git status


##quando aparecer a mensagem em vermelho informando a presença de novas mudanças:
git add .


##após aparecer a mensagem em verder executar os seguintes comando em sequencia:
git commit -m "(comentario)"  ##deve colocar um comentário pra identificar a mudança
git pull origin master
 ##baixa a versão mais atualizada atual
git push origin status ##envia as mudanças
